CREATE TABLE Members (
    MemberID INT PRIMARY KEY,
    Name VARCHAR(255),
    DateOfBirth DATE,
    Gender VARCHAR(50),
    ContactInfo VARCHAR(255),
    HealthMetrics VARCHAR(255),
    FitnessGoals VARCHAR(255)
);
CREATE TABLE Trainers (
    TrainerID INT PRIMARY KEY,
    Name VARCHAR(255),
    Qualifications VARCHAR(255)
);
CREATE TABLE Staff (
    StaffID INT PRIMARY KEY,
    Name VARCHAR(255),
    Role VARCHAR(100),
    ContactInfo VARCHAR(255)
);
CREATE TABLE GroupClasses (
    ClassID INT PRIMARY KEY,
    Name VARCHAR(255),
    TrainerID INT,
    RoomID INT,
    Schedule TEXT,
    MaxCapacity INT,
    FOREIGN KEY (TrainerID) REFERENCES Trainers(TrainerID),
    FOREIGN KEY (RoomID) REFERENCES Room(RoomID)
);
CREATE TABLE Equipment (
    EquipmentID INT PRIMARY KEY,
    Type VARCHAR(100),
    Status VARCHAR(50),
    LastMaintenanceDate DATE
);
CREATE TABLE Room (
    RoomID INT PRIMARY KEY,
    Name VARCHAR(255),
    Capacity INT,
    Type VARCHAR(100)
);
CREATE TABLE Event (
    EventID INT PRIMARY KEY,
    Name VARCHAR(255),
    Description TEXT,
    DateTime DATETIME,
    Location VARCHAR(255)
);
CREATE TABLE MemberClassParticipation (
    MemberID INT,
    ClassID INT,
    DateJoined DATE,
    PRIMARY KEY (MemberID, ClassID),
    FOREIGN KEY (MemberID) REFERENCES Members(MemberID),
    FOREIGN KEY (ClassID) REFERENCES GroupClasses(ClassID)
);
CREATE TABLE Billing (
    BillID INT PRIMARY KEY,
    MemberID INT,
    Amount DECIMAL(10, 2),
    Date DATE,
    PaymentMethod VARCHAR(100),
    FOREIGN KEY (MemberID) REFERENCES Members(MemberID)
);
CREATE TABLE LoyaltyPoints (
    MemberID INT PRIMARY KEY,
    PointsEarned INT,
    FOREIGN KEY (MemberID) REFERENCES Members(MemberID)
);
CREATE TABLE ProgressNotes (
    NoteID INT PRIMARY KEY,
    ClassID INT,
    TrainerID INT,
    Content TEXT,
    Date DATE,
    FOREIGN KEY (ClassID) REFERENCES GroupClasses(ClassID),
    FOREIGN KEY (TrainerID) REFERENCES Trainers(TrainerID)
);
CREATE TABLE Feedback (
    FeedbackID INT PRIMARY KEY,
    MemberID INT,
    ClassID INT,
    Content TEXT,
    Date DATE,
    FOREIGN KEY (MemberID) REFERENCES Members(MemberID),
    FOREIGN KEY (ClassID) REFERENCES GroupClasses(ClassID)
);
CREATE TABLE Notifications (
    NotificationID INT PRIMARY KEY,
    MemberID INT,
    Content TEXT,
    DateSent DATE,
    FOREIGN KEY (MemberID) REFERENCES Members(MemberID)
);
CREATE TABLE MemberEventParticipation (
    MemberID INT,
    EventID INT,
    DateJoined DATE,
    PRIMARY KEY (MemberID, EventID),
    FOREIGN KEY (MemberID) REFERENCES Members(MemberID),
    FOREIGN KEY (EventID) REFERENCES Event(EventID)
);
